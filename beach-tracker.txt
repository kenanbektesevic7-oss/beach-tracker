<script>
// ── STATE ──
var state = {today:[],season:[],stock:[],goal:300};
try{ var s=localStorage.getItem('solo2020'); if(s) state=JSON.parse(s); }catch(e){}

function save(){try{localStorage.setItem('solo2020',JSON.stringify(state));}catch(e){}}

// DATE
document.getElementById('datebadge').textContent=new Date().toLocaleDateString('en-GB',{weekday:'short',day:'numeric',month:'short'});

// ── TABS ──
function switchTab(name){
  ['calc','today','stock','season'].forEach(function(t){
    document.getElementById('panel-'+t).classList.remove('active');
    document.getElementById('tab-'+t).classList.remove('active');
  });
  document.getElementById('panel-'+name).classList.add('active');
  document.getElementById('tab-'+name).classList.add('active');
  if(name==='today') renderToday();
  if(name==='stock') renderStock();
  if(name==='season') renderSeason();
}

// ── CALC ──
function calcProfit(){
  var buy=parseFloat(document.getElementById('c-buy').value)||0;
  var sell=parseFloat(document.getElementById('c-sell').value)||0;
  var qty=parseInt(document.getElementById('c-qty').value)||0;
  var box=document.getElementById('calc-result');
  if(!buy&&!sell&&!qty){box.classList.remove('show');return;}
  box.classList.add('show');
  document.getElementById('r-cost').textContent='€'+(buy*qty).toFixed(2);
  document.getElementById('r-revenue').textContent='€'+(sell*qty).toFixed(2);
  document.getElementById('r-margin').textContent='€'+(sell-buy).toFixed(2);
  document.getElementById('r-profit').textContent='€'+((sell-buy)*qty).toFixed(2);
}

function logSale(){
  var buy=parseFloat(document.getElementById('c-buy').value)||0;
  var sell=parseFloat(document.getElementById('c-sell').value)||0;
  var qty=parseInt(document.getElementById('c-qty').value)||0;
  var name=document.getElementById('c-name').value||'Item';
  var round=parseInt(document.getElementById('c-round').value)||1;
  if(!sell||!qty){alert('Enter sell price and quantity');return;}
  state.today.push({name:name,buy:buy,sell:sell,qty:qty,round:round,
    profit:(sell-buy)*qty,revenue:sell*qty,
    time:new Date().toLocaleTimeString('en-GB',{hour:'2-digit',minute:'2-digit'})});
  save();
  document.getElementById('c-buy').value='';
  document.getElementById('c-sell').value='';
  document.getElementById('c-qty').value='';
  document.getElementById('c-name').value='';
  document.getElementById('c-round').value='';
  document.getElementById('calc-result').classList.remove('show');
  alert('Sale logged!');
}

// ── TODAY ──
function removeSale(i){
  if(!confirm('Remove this sale?')) return;
  state.today.splice(i,1);
  save();
  renderToday();
}

function renderToday(){
  var log=document.getElementById('today-log');
  if(!state.today.length){
    log.innerHTML='<div class="empty"><div class="ei">🏖️</div>No sales yet.<br>Use Calc tab to log sales.</div>';
    document.getElementById('t-profit').textContent='€0.00';
    document.getElementById('t-revenue').textContent='€0.00';
    document.getElementById('t-sold').textContent='0';
    document.getElementById('t-rounds').textContent='0';
    return;
  }
  var totalProfit=0,totalRevenue=0,totalSold=0,roundSet={};
  for(var j=0;j<state.today.length;j++){
    totalProfit+=state.today[j].profit;
    totalRevenue+=state.today[j].revenue;
    totalSold+=state.today[j].qty;
    roundSet[state.today[j].round]=1;
  }
  document.getElementById('t-profit').textContent='€'+totalProfit.toFixed(2);
  document.getElementById('t-revenue').textContent='€'+totalRevenue.toFixed(2);
  document.getElementById('t-sold').textContent=totalSold;
  document.getElementById('t-rounds').textContent=Object.keys(roundSet).length;
  var html='';
  for(var i=0;i<state.today.length;i++){
    var s=state.today[i];
    html+='<div class="li">';
    html+='<div class="li-top">';
    html+='<div><div class="li-name">'+s.name+'</div><div class="li-meta">Round '+s.round+' · '+s.qty+' units · '+s.time+'</div></div>';
    html+='<div><div class="li-profit">+€'+s.profit.toFixed(2)+'</div><div class="li-sub">€'+s.sell+' each</div></div>';
    html+='</div>';
    html+='<button class="rm-btn" onclick="removeSale('+i+')">🗑 REMOVE SALE</button>';
    html+='</div>';
  }
  log.innerHTML=html;
}

function endDay(){
  if(!state.today.length){alert('No sales to save');return;}
  if(!confirm('End today and save to season?')) return;
  var p=0,sold=0,rev=0;
  for(var i=0;i<state.today.length;i++){p+=state.today[i].profit;sold+=state.today[i].qty;rev+=state.today[i].revenue;}
  state.season.push({date:new Date().toLocaleDateString('en-GB',{day:'numeric',month:'short'}),profit:p,sold:sold,revenue:rev,sales:state.today.length});
  state.today=[];
  save();
  alert('Day saved to season!');
  renderToday();
}

// ── STOCK ──
function addStock(){
  var name=document.getElementById('i-name').value;
  var qty=parseInt(document.getElementById('i-qty').value)||0;
  var buy=parseFloat(document.getElementById('i-buy').value)||0;
  var sell=parseFloat(document.getElementById('i-sell').value)||0;
  if(!name||!qty){alert('Enter item name and quantity');return;}
  state.stock.push({name:name,qty:qty,remaining:qty,buy:buy,sell:sell});
  save();
  document.getElementById('i-name').value='';
  document.getElementById('i-qty').value='';
  document.getElementById('i-buy').value='';
  document.getElementById('i-sell').value='';
  renderStock();
}

function sellStock(i){
  var input=document.getElementById('sq-'+i);
  var qty=parseInt(input.value)||0;
  var s=state.stock[i];
  if(!qty||qty<1){alert('Enter how many you sold');return;}
  if(qty>s.remaining){alert('Only '+s.remaining+' left in stock');return;}
  state.stock[i].remaining-=qty;
  state.today.push({name:s.name,buy:s.buy,sell:s.sell,qty:qty,round:1,
    profit:(s.sell-s.buy)*qty,revenue:s.sell*qty,
    time:new Date().toLocaleTimeString('en-GB',{hour:'2-digit',minute:'2-digit'})});
  if(state.stock[i].remaining===0) state.stock.splice(i,1);
  save();
  renderStock();
  alert(qty+'x '+s.name+' sold and logged to Today!');
}

function removeStock(i){
  if(!confirm('Remove "'+state.stock[i].name+'" from stock?')) return;
  state.stock.splice(i,1);
  save();
  renderStock();
}

function renderStock(){
  var list=document.getElementById('stock-list');
  if(!state.stock.length){
    list.innerHTML='<div class="empty"><div class="ei">📦</div>No stock added yet.</div>';
    return;
  }
  var html='';
  for(var i=0;i<state.stock.length;i++){
    var s=state.stock[i];
    var pct=Math.round((s.remaining/s.qty)*100);
    var col=pct>50?'var(--accent)':pct>20?'var(--yellow)':'var(--red)';
    html+='<div class="li">';
    html+='<div class="li-top">';
    html+='<div><div class="li-name">'+s.name+'</div><div class="li-meta">Buy €'+s.buy+' · Sell €'+s.sell+'</div></div>';
    html+='<div><div class="li-profit" style="color:'+col+'">'+s.remaining+'/'+s.qty+'</div><div class="li-sub">remaining</div></div>';
    html+='</div>';
    html+='<div class="sbar-wrap"><div class="sbar-fill" style="width:'+pct+'%;background:'+col+'"></div></div>';
    html+='<div class="sell-row" style="margin-top:10px">';
    html+='<input type="number" id="sq-'+i+'" placeholder="Qty sold" min="1" style="padding:8px 10px;font-size:14px">';
    html+='<button class="mini-btn mini-btn-a" onclick="sellStock('+i+')">✓ Sold</button>';
    html+='<button class="mini-btn" style="background:rgba(255,77,106,.15);color:var(--red);border:1px solid var(--red)" onclick="removeStock('+i+')">🗑</button>';
    html+='</div>';
    html+='</div>';
  }
  list.innerHTML=html;
}

// ── SEASON ──
function renderSeason(){
  var total=0,units=0;
  for(var i=0;i<state.season.length;i++){total+=state.season[i].profit;units+=state.season[i].sold;}
  document.getElementById('s-total').textContent='€'+total.toFixed(2);
  document.getElementById('s-days').textContent=state.season.length;
  document.getElementById('s-units').textContent=units;
  var goal=state.goal||300;
  var pct=Math.min((total/goal)*100,100);
  document.getElementById('goal-bar').style.width=pct+'%';
  document.getElementById('goal-pct').textContent=pct.toFixed(0)+'%';
  document.getElementById('goal-display').textContent='Goal: €'+goal;
  document.getElementById('remaining').textContent='€'+Math.max(goal-total,0).toFixed(2);
  var daysLeft=parseInt(document.getElementById('days-left').value)||60;
  var daysRemaining=Math.max(daysLeft-state.season.length,1);
  document.getElementById('per-day').textContent='€'+(Math.max(goal-total,0)/daysRemaining).toFixed(2);
  var log=document.getElementById('season-log');
  if(!state.season.length){
    log.innerHTML='<div class="empty"><div class="ei">📅</div>No days saved yet.</div>';
    return;
  }
  var html='';
  for(var i=state.season.length-1;i>=0;i--){
    var d=state.season[i];
    html+='<div class="li"><div class="li-top">';
    html+='<div><div class="li-name">'+d.date+'</div><div class="li-meta">'+d.sold+' units · '+d.sales+' sales</div></div>';
    html+='<div><div class="li-profit">+€'+d.profit.toFixed(2)+'</div><div class="li-sub">€'+d.revenue.toFixed(2)+' rev</div></div>';
    html+='</div></div>';
  }
  log.innerHTML=html;
}

function updateGoal(){
  state.goal=parseFloat(document.getElementById('goal-input').value)||300;
  save();
  renderSeason();
}

function resetSeason(){
  if(!confirm('Reset entire season? Cannot be undone.')) return;
  state={today:[],season:[],stock:[],goal:300};
  save();
  renderToday();
